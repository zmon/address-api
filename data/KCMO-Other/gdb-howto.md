We are only intrested in the `CREATE FOREIGN TABLE` section

````
CREATE SERVER myserver
  FOREIGN DATA WRAPPER ogr_fdw
  OPTIONS (
    datasource 'Other.gdb',
    format 'OpenFileGDB' );

CREATE FOREIGN TABLE vacantparcels (
  fid bigint,
  shape Geometry(MultiPolygon),
  kivapin varchar,
  apn varchar,
  address varchar,
  addr integer,
  fraction varchar,
  prefix varchar,
  street varchar,
  street_type varchar,
  suite varchar,
  own_name varchar,
  own_addr varchar,
  own_city varchar,
  own_state varchar,
  own_zip varchar,
  zoning varchar,
  shape_length real,
  shape_area real
) SERVER myserver
OPTIONS (layer 'VacantParcels');
````
